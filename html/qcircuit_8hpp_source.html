123  void decomposePsi(Args args = Args::global()){
124  ITensor U,S,V;
125 
126  // fetch nodelist in index "first"
127  auto list = topology.neighborsOf(cursor.first);
128 
129  /* get link index corresponding to cursor position
130  * (this can be combined with the std::remove_if below)
131  */
132  auto second_itr = std::find_if(list.begin(), list.end(),
133  [&](auto x) {return x.site == cursor.second;});
134  auto link_ind = (*second_itr).link;
135 
136  // remove the element which connects to node second
137  auto pend = std::remove_if(list.begin(), list.end(), [&](const auto& t){return t.site == cursor.second;});
138  std::vector<CircuitTopology::Neighbor> newlist;
139  for(auto it = list.begin(); it != pend; ++it){
140  newlist.push_back(*it);
141  }
142  switch(newlist.size()) {
143  case 0:
144  U = ITensor(s[cursor.first]);
145  break;
146  case 1:
147  U = ITensor(s[cursor.first], a[newlist[0].link]);
148  break;
149  case 2:
150  U = ITensor(s[cursor.first], a[newlist[0].link], a[newlist[1].link]);
151  break;
152  default:
153  assert(false);
154  }
155 
156  Spectrum spec = svd(Psi, U, S, V, args);
157  a[link_ind] = commonIndex(U,S);
158  S /= norm(S); //normalize
159  M[cursor.first] = U;
160  M[cursor.second] = S*V;
161  }

164  Spectrum shiftCursorTo(size_t ind, const Args& args = Args::global()) {
165  assert(ind != cursor.first);
166  assert(ind != cursor.second);
167 
168  //search ind from list
169  int flag = 0; //0 not found 1 connected with cursor.first 2 connected with cursor.second
170 
171  //first
172  for(auto&& i : topology.neighborsOf(cursor.first)) {
173  if(i.site == ind){
174  flag = 1;
175  break;
176  }
177  }
178  //second
179  for(auto&& i : topology.neighborsOf(cursor.second)) {
180  if(i.site == ind){
181  flag = 2;
182  break;
183  }
184  }
185 
186  Spectrum spec;
187 
188  if(flag == 1) {
189  ITensor U,S,V;
190 
191  //fetch nodelist in index "second"
192  auto list = this->topology.neighborsOf(cursor.second);
193  //get link index
194  size_t link_ind = 0;
195  for(auto&& i : list){
196  if(i.site == cursor.first){
197  link_ind = i.link;
198  }
199  }
200  //remove the element which connects to node first
201  auto pend = std::remove_if(list.begin(), list.end(), [&](const auto& t){return t.site == cursor.first;});
202  std::vector<CircuitTopology::Neighbor> newlist;
203  for(auto it = list.begin(); it != pend; ++it){
204  newlist.push_back(*it);
205  }
206  switch(newlist.size()){
207  case 0:
208  V = ITensor(s[cursor.second]);
209  break;
210  case 1:

212  break;
213  case 2:

215  break;
216  default:

218  }

221 
222  a[link_ind] = commonIndex(S,V);
223  S /= norm(S); //normalize
224  M[cursor.second] = V;
225  Psi = M[ind]*U*S;
226 
227  cursor.second = cursor.first;
228  cursor.first = ind;
229  } else if(flag == 2) {

244  for(auto it = list.begin(); it != pend; ++it){
245  newlist.push_back(*it);
246  }
247  switch(newlist.size()){
248  case 0:
249  U = ITensor(s[cursor.first]);

251  case 1:
252  U = ITensor(s[cursor.first], a[newlist[0].link]);
253  break;
254  case 2:
255  U = ITensor(s[cursor.first], a[newlist[0].link], a[newlist[1].link]);
256  break;
257  default:
258  assert(false);
259  }

261  spec = svd(Psi, U, S, V, args);
262 
263  a[link_ind] = commonIndex(U,S);
264  S /= norm(S); //normalize
265  M[cursor.first] = U;
266  Psi = S*V*M[ind];

268  cursor.first = cursor.second;

331  double probabilityOfZero(size_t site) const {
332  std::vector<ITensor> op;
333  op.reserve(this->size());
334  for(size_t i = 0;i < this->size();i++) {
335  if(i == site) {
336  op.push_back(generateTensorOp(Proj_0(i)));
337  } else {
338  op.push_back(generateTensorOp(Id(i)));
339  }
340  }
341 
342  /*
343  * <psi|Proj_0(i)|psi> = |w_{i,0}|^2 is real by definition,
344  * where w_{i,0} is the coefficient corresponding to the base |....0....>.
345  * (ith qubit) ^
346  */
347  return std::real(overlap(*this, op, *this));
348  }

351  int observeQubit(size_t site, const Args& args = Args::global()) {
352  auto prob0 = probabilityOfZero(site);
353 
354  std::uniform_real_distribution<> dist(0.0, 1.0);
355  int state = (dist(random_engine) < prob0) ? 0 : 1; // measurement
356 
357  size_t neighbor = topology.neighborsOf(site)[0].site; // Dummy site to which Id operator is applied.
358  if(state == 0) {
359  apply(Proj_0(site), Id(neighbor), args);
360  } else {
361  apply(Proj_1(site), Id(neighbor), args);
362  }
363  this->normalize();
364 
365  return state;
366  }

427  inline Cplx overlap(QCircuit circuit1,
428  const std::vector<ITensor>& op,
429  QCircuit circuit2,
430  const Args& args) {
431  /* Prototype declaration of this function is placed at the top. */
432  assert(op.size() == circuit1.size() && op.size() == circuit2.size());
433 
434  circuit1.decomposePsi(args);
435  circuit2.decomposePsi(args);
436 
437  circuit2.primeAll();
438 
439  ITensor ret_t = dag(circuit1.Mref(0))*op[0]*circuit2.Mref(0);
440  for(size_t i = 1; i < circuit1.size(); i++) {
441  //reduction
442  ret_t = dag(circuit1.Mref(i))*op[i]*ret_t*circuit2.Mref(i);
443  }
444 
445  return ret_t.cplx();
446  }
