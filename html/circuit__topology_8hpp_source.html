64  std::vector<size_t> getRoute(std::pair<size_t, size_t> origin, std::pair<size_t, size_t> destination) const {
65  static const int NOT_YET_REACHED = -1;
66  std::vector<int> back_to(num_bits, NOT_YET_REACHED);
67  // back_to[i] points which neighboring site we go back at ith site
68  // to reach `origin` in the shortest path.
69 
70  // if not necessary to move
71  if((origin.first == destination.first && origin.second == destination.second) ||
72  (origin.first == destination.second && origin.second == destination.first)) {
73  return std::vector<size_t>();
74  }
75 
76  std::queue<size_t> queue;
77  queue.push(origin.first);
78  back_to[origin.first] = origin.first;
79  queue.push(origin.second);
80  back_to[origin.second] = origin.second;
81 
82  while(!queue.empty()) {
83  auto site = queue.front();
84  queue.pop(); // pop() returns nothing.
85 
86  if(site == destination.first || site == destination.second) {
87  break;
88  }
89 
90  for(auto neighbor : neighbors_list[site]) {
91  if(back_to[neighbor.site] == NOT_YET_REACHED) {
92  back_to[neighbor.site] = site;
93  queue.push(neighbor.site);
94  }
95  }
96  }
97 
98  size_t site;
99  std::vector<size_t> result;
100  if(back_to[destination.first] != NOT_YET_REACHED) {
101  result.push_back(destination.second);
102  site = destination.first;
103  } else if(back_to[destination.second] != NOT_YET_REACHED) {
104  result.push_back(destination.first);
105  site = destination.second;
106  } else {
107  // TODO: assert or throw Exception.
108  // If reach this else branch, given sites are unreachable.
109  }
110 
111  while(site != origin.first && site != origin.second) {
112  result.push_back(site);
113  site = back_to[site];
114  }
115  std::reverse(result.begin(), result.end());
116  return result;
117  }
